257  private static void GenerateUriParameters(HelpPageApiModel apiModel, ModelDescriptionGenerator modelGenerator)
258  {
259  ApiDescription apiDescription = apiModel.ApiDescription;
260  foreach (ApiParameterDescription apiParameter in apiDescription.ParameterDescriptions)
261  {
262  if (apiParameter.Source == ApiParameterSource.FromUri)
263  {
264  HttpParameterDescriptor parameterDescriptor = apiParameter.ParameterDescriptor;
265  Type parameterType = null;
266  ModelDescription typeDescription = null;
267  ComplexTypeModelDescription complexTypeDescription = null;
268  if (parameterDescriptor != null)
269  {
270  parameterType = parameterDescriptor.ParameterType;
271  typeDescription = modelGenerator.GetOrCreateModelDescription(parameterType);
272  complexTypeDescription = typeDescription as ComplexTypeModelDescription;
273  }
274 
275  // Example:
276  // [TypeConverter(typeof(PointConverter))]
277  // public class Point
278  // {
279  // public Point(int x, int y)
280  // {
281  // X = x;
282  // Y = y;
283  // }
284  // public int X { get; set; }
285  // public int Y { get; set; }
286  // }
287  // Class Point is bindable with a TypeConverter, so Point will be added to UriParameters collection.
288  //
289  // public class Point
290  // {
291  // public int X { get; set; }
292  // public int Y { get; set; }
293  // }
294  // Regular complex class Point will have properties X and Y added to UriParameters collection.
295  if (complexTypeDescription != null
296  && !IsBindableWithTypeConverter(parameterType))
297  {
298  foreach (ParameterDescription uriParameter in complexTypeDescription.Properties)
299  {
300  apiModel.UriParameters.Add(uriParameter);
301  }
302  }
303  else if (parameterDescriptor != null)
304  {
305  ParameterDescription uriParameter =
306  AddParameterDescription(apiModel, apiParameter, typeDescription);
307 
308  if (!parameterDescriptor.IsOptional)
309  {
310  uriParameter.Annotations.Add(new ParameterAnnotation() { Documentation = "Required" });
311  }
312 
313  object defaultValue = parameterDescriptor.DefaultValue;
314  if (defaultValue != null)
315  {
316  uriParameter.Annotations.Add(new ParameterAnnotation() { Documentation = "Default value is " + Convert.ToString(defaultValue, CultureInfo.InvariantCulture) });
317  }
318  }
319  else
320  {
321  Debug.Assert(parameterDescriptor == null);
322 
323  // If parameterDescriptor is null, this is an undeclared route parameter which only occurs
324  // when source is FromUri. Ignored in request model and among resource parameters but listed
325  // as a simple string here.
326  ModelDescription modelDescription = modelGenerator.GetOrCreateModelDescription(typeof(string));
327  AddParameterDescription(apiModel, apiParameter, modelDescription);
328  }
329  }
330  }
331  }

416  private static bool TryGetResourceParameter(ApiDescription apiDescription, HttpConfiguration config, out ApiParameterDescription parameterDescription, out Type resourceType)
417  {
418  parameterDescription = apiDescription.ParameterDescriptions.FirstOrDefault(
419  p => p.Source == ApiParameterSource.FromBody ||
420  (p.ParameterDescriptor != null && p.ParameterDescriptor.ParameterType == typeof(HttpRequestMessage)));
421 
422  if (parameterDescription == null)
423  {
424  resourceType = null;
425  return false;
426  }
427 
428  resourceType = parameterDescription.ParameterDescriptor.ParameterType;
429 
430  if (resourceType == typeof(HttpRequestMessage))
431  {
432  HelpPageSampleGenerator sampleGenerator = config.GetHelpPageSampleGenerator();
433  resourceType = sampleGenerator.ResolveHttpRequestMessageType(apiDescription);
434  }
435 
436  if (resourceType == null)
437  {
438  parameterDescription = null;
439  return false;
440  }
441 
442  return true;
443  }
